126 word ladder II

Published: 周四 08 九月 2016
By bingchen

In leetcode.

Problem

Given two words (beginWord and endWord), and a dictionary's word list, find all shortest transformation sequence(s) from beginWord to endWord, such that: 1. Only one letter can be changed at a time 2. Each intermediate word must exist in the word list
For example,
Given:
beginWord = "hit"
endWord = "cog"
wordList = ["hot","dot","dog","lot","log"]
Return 

  [
    ["hit","hot","dot","dog","cog"],
    ["hit","hot","lot","log","cog"]
  ]

Note: All words have the same length. All words contain only lowercase alphabetic characters.

Thoughts

题目的本质是寻找最短路径并打印，但由于所有的路径长度都为1，可以使用bfs或者dfs来寻找
我最开始的想法是先将所有的单词组成一张图，然后利用bfs来搜索路径。这样的算法可以勉强ac，不过 不是每次提交都过，可能是因为unordered_set里面采用了随机函数的原因吧。 后来发现并不需要建立图，因为两个节点之间是否联通可以直接判断。

Solution

  1. 利用bfs从初始单词出发，记录每个单词从初始单词开始到它自己的路径。建立一个类似于树的结构，如下图：
    使用unordered_map<string, vector<string>> next表示树，next中存相应字符串在树中的孩子。
  2. 利用dfs遍历树，并打印路径
  3. 为了加快bfs的搜索速度，可以从初始和结尾单词同时开始搜索，每次都遍历较小的那个队列， 当一个队列搜索到出现在另一个队列的元素的时候，就停止但是需要注意从结尾单词搜索的时候，搜索 到的单词实际是当前单词的父亲，需要一个变量来记录是从头搜索的还是从尾搜索的。
  4. 搜索下一个单词的时候不要遍历所有的单词，而是采用字母遍历的方式，即枚举所有的与当前单词相差 一个字母的单词，然后看他是否在单词列表中。
  5. bfs要防止重复搜索，最开始的做法是使用一个单独的变量存储已经访问过的单词，后来发现可以简单的将 已经访问过的单词移除。

pseudocode & code

class Solution {
public:
    unordered_map<string, vector<string>> next;
    vector<vector<string>> findLadders(string beginWord, string endWord, unordered_set<string> &wordList) {
        //1. 建立从beginWord到endWord的最短路径树,存在next中
        unordered_set<string> headSet, tailSet, *smallerSet, *largerSet;
        headSet.insert(beginWord);
        tailSet.insert(endWord);
        wordList.erase(beginWord);
        wordList.erase(endWord);
        bool forward = true;
        bool connected = false;
        while (!connected) {
            // 选择较小的队列进行展开
            if (headSet.size() > tailSet.size()) {
                smallerSet = &tailSet;
                largerSet = &headSet;
                forward = false;
            } else {
                largerSet = &tailSet;
                smallerSet = &headSet;
                forward = true;
            }
            unordered_set<string> newSet;
            for (string str: *smallerSet) {   //遍历较小的队列
                for (int i = 0 ; i < str.size() ; ++i) {
                    string tmp = str;   //error: 放错过位置，放在了下层循环中
                    for (char c = 'a' ; c <= 'z' ; ++c) {   //字符遍历
                        tmp[i] = c;
                        if (largerSet->count(tmp)) { //连接上了
                            connected = true;
                            if (forward) {
                                next[str].push_back(tmp);
                            } else {
                                next[tmp].push_back(str);
                            }
                        }
                        if (wordList.count(tmp)) {
                            if (forward) {
                                next[str].push_back(tmp);
                            } else {
                                next[tmp].push_back(str);
                            }
                            newSet.insert(tmp);
                        }
                    }
                }
            }
            if (newSet.empty()) {    //无法到达
                break;
            }
            for (string str : newSet) {
                wordList.erase(str);
            }
            swap(newSet, *smallerSet);
        }
        //2. 依据next，利用dfs，得到结果，存在ret中
        vector<vector<string>> ret;
        vector<string> partRet;
        partRet.push_back(beginWord);     //遗漏过
        dfs(beginWord, endWord, partRet, ret);
        return ret;
    }

    void dfs(string cur, string& endStr, vector<string>& partRet, vector<vector<string>>& ret) {
        if (cur == endStr) {
            ret.push_back(partRet);
            return;
        }

        for (string str : next[cur]) {
            partRet.push_back(str);
            dfs(str, endStr, partRet, ret);
            partRet.pop_back();
        }
    }
};
